// Распознаёт ведомость по присоединённому файлу с постобработкой ответа VLM.

// Параметры:
//  Ведомость - ДокументСсылка.GDS_ЗаправочнаяВедомость
//  Двухпроходно - Булево - Истина: отдельно шапка/подвал и таблица
//
// Возвращаемое значение:
//  Структура - Данные, ДанныеJSON, Предупреждения, ОтветыAPI, Двухпроходно
//
Функция РаспознатьВедомость(Ведомость, Двухпроходно = Истина) Экспорт
	
	ИмяТипа = ИмяТипаВедомостиДляПромптаVLM(Ведомость.ТипВедомости);
	
	Файлы = Новый Массив;
	РаботаСФайлами.ЗаполнитьПрисоединенныеФайлыКОбъекту(Ведомость, Файлы);
	
	Если Файлы.Количество() = 0 Тогда
		ВызватьИсключение НСтр("ru = 'Нет присоединённых файлов у ведомости.'");
	КонецЕсли;
	
	Возврат РаспознатьВедомостьПоФайлу(Файлы[0], ИмяТипа, Двухпроходно);

// Распознаёт скан ведомости по ссылке на присоединённый файл.
//
// Параметры:
//  Файл - СправочникСсылка.Файлы
//  ИмяТипа - Строка - zapravka | prihodnaya | perelivnaya
//  Двухпроходно - Булево
//
// Возвращаемое значение:
//  Структура - Данные, ДанныеJSON, Предупреждения, ОтветыAPI, Двухпроходно
//
Функция РаспознатьВедомостьПоФайлу(Файл, ИмяТипа, Двухпроходно = Истина) Экспорт
	
	ОтветыAPI = Новый Массив;
	
	Если Двухпроходно Тогда
		ДанныеVLM = РаспознатьДвухпроходно(Файл, ИмяТипа, ОтветыAPI);
	Иначе
		ДанныеVLM = РаспознатьОднопроходно(Файл, ИмяТипа, ОтветыAPI);
	КонецЕсли;
	
	ДанныеVLM = НормализоватьОтветVLM(ДанныеVLM, ИмяТипа);
	Предупреждения = ПроверитьИИсправитьОтветVLM(ДанныеVLM, ИмяТипа);
	
	Результат = Новый Структура;
	Результат.Вставить("Данные", ДанныеVLM);
	Результат.Вставить("ДанныеJSON", ЗаписатьСтруктуруВJSON(ДанныеVLM));
	Результат.Вставить("Предупреждения", Предупреждения);
	Результат.Вставить("ОтветыAPI", ОтветыAPI);
	Результат.Вставить("Двухпроходно", Двухпроходно);
	
	Возврат Результат;
	
КонецФункции

// Выполняет запрос к VLM и возвращает разобранный JSON из content.
//
// Параметры:
//  Файл - СправочникСсылка.Файлы
//  Промпт - Структура - Системный, Пользовательский
//
// Возвращаемое значение:
//  Структура - ОтветAPI, Данные
//
Функция ЗапросРазспознаванияИИ(Файл, Промпт) Экспорт
	
	ДвоичныеДанные = РаботаСФайлами.ДвоичныеДанныеФайла(Файл);
	Base64Изображения = Base64Строка(ДвоичныеДанные);
	
	ОтветAPI = ВыполнитьЗапросVLM(Base64Изображения, Промпт);
	Данные = ИзвлечьДанныеИзОтветаVLM(ОтветAPI);
	
	Результат = Новый Структура;
	Результат.Вставить("ОтветAPI", ОтветAPI);
	Результат.Вставить("Данные", Данные);
	
	Возврат Результат;

Функция ВыполнитьЗапросVLM(Base64Изображения, Промпт)
	
	МассивСообщений = Новый Массив;
	
	СообщениеСистемы = Новый Структура("role, content", "system", Промпт.Системный);
	МассивСообщений.Добавить(СообщениеСистемы);
	
	МассивContent = Новый Массив;
	
	ТекстовыйБлок = Новый Структура;
	ТекстовыйБлок.Вставить("type", "text");
	ТекстовыйБлок.Вставить("text", Промпт.Пользовательский);
	МассивContent.Добавить(ТекстовыйБлок);
	
	БлокИзображения = Новый Структура;
	БлокИзображения.Вставить("type", "image_url");
	СтруктураURL = Новый Структура;
	СтруктураURL.Вставить("url", "data:image/jpeg;base64," + Base64Изображения);
	БлокИзображения.Вставить("image_url", СтруктураURL);
	МассивContent.Добавить(БлокИзображения);
	
	СообщениеПользователя = Новый Структура("role, content", "user", МассивContent);
	МассивСообщений.Добавить(СообщениеПользователя);
	
	СтруктураЗапроса = Новый Структура;
	СтруктураЗапроса.Вставить("model", "qwen3-vl-8b-instruct");
	СтруктураЗапроса.Вставить("messages", МассивСообщений);
	СтруктураЗапроса.Вставить("max_tokens", 4096);
	СтруктураЗапроса.Вставить("temperature", 0);
	СтруктураЗапроса.Вставить("top_p", 0.8);
	
	ДополнительныеПараметры = GDS_HTTPКоннектор.НовыеПараметры();
	ДополнительныеПараметры.Заголовки.Вставить("Content-Type", "application/json");
	ДополнительныеПараметры.Заголовки.Вставить("Authorization", "Bearer " + ТокенДоступаVLM());
	ДополнительныеПараметры.Json = СтруктураЗапроса;
	ДополнительныеПараметры.Таймаут = 600;
	
	HTTPОтвет = GDS_HTTPКоннектор.Post(АдресСервисаVLM(), , ДополнительныеПараметры);
	
	Если HTTPОтвет.КодСостояния <> 200 Тогда
		ТелоОшибки = GDS_HTTPКоннектор.КакТекст(HTTPОтвет);
		ВызватьИсключение СтрШаблон(
			НСтр("ru = 'Ошибка HTTP %1 при запросе к VLM: %2'"),
			HTTPОтвет.КодСостояния,
			ТелоОшибки);
	КонецЕсли;
	
	Возврат GDS_HTTPКоннектор.КакJson(HTTPОтвет, ПараметрыПреобразованияJSONПоУмолчанию());

Функция ИзвлечьДанныеИзОтветаVLM(ОтветAPI)
	
	Если ОтветAPI = Неопределено Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Choices = ПолучитьСвойствоJSON(ОтветAPI, "choices");
	Если Choices = Неопределено Или ТипЗнч(Choices) <> Тип("Массив") Или Choices.Количество() = 0 Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Message = ПолучитьСвойствоJSON(Choices[0], "message");
	Content = ПолучитьСвойствоJSON(Message, "content");
	
	Если Content = Неопределено Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Если ТипЗнч(Content) = Тип("Структура") Или ТипЗнч(Content) = Тип("Соответствие") Тогда
		Возврат JSONВСтруктуру(Content);
	КонецЕсли;
	
	ТекстJSON = СокрЛП(Строка(Content));
	ТекстJSON = УбратьОбрамлениеMarkdownJSON(ТекстJSON);
	
	Если ПустаяСтрока(ТекстJSON) Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Возврат РазобратьJSONСтроку(ТекстJSON);
	
КонецФункции

Функция JSONВСтруктуру(ОбъектJSON)
	
	Если ТипЗнч(ОбъектJSON) = Тип("Структура") Тогда
		Возврат ОбъектJSON;
	КонецЕсли;
	
	Если ТипЗнч(ОбъектJSON) = Тип("Соответствие") Тогда
		Возврат РазобратьJSONСтроку(GDS_HTTPКоннектор.ОбъектВJson(ОбъектJSON));
	КонецЕсли;
	
	Возврат Неопределено;
	
КонецФункции

// Универсальное чтение свойства из Структуры или Соответствия (ответ GDS_HTTPКоннектор.КакJson).
//
Функция ПолучитьСвойствоJSON(ОбъектJSON, ИмяСвойства)
	
	Если ОбъектJSON = Неопределено Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Если ТипЗнч(ОбъектJSON) = Тип("Структура") Тогда
		Значение = Неопределено;
		Если ОбъектJSON.Свойство(ИмяСвойства, Значение) Тогда
			Возврат Значение;
		КонецЕсли;
		Возврат Неопределено;
	ИначеЕсли ТипЗнч(ОбъектJSON) = Тип("Соответствие") Тогда
		Возврат ОбъектJSON.Получить(ИмяСвойства);
	КонецЕсли;
	
	Возврат Неопределено;
	
КонецФункции

Функция УбратьОбрамлениеMarkdownJSON(ТекстJSON)
	
	Если Не СтрНачинаетсяС(ТекстJSON, "```") Тогда
		Возврат ТекстJSON;
	КонецЕсли;
	
	ПозицияПеревода = СтрНайти(ТекстJSON, Символы.ПС);
	Если ПозицияПеревода = 0 Тогда
		Возврат ТекстJSON;
	КонецЕсли;
	
	ТекстJSON = Сред(ТекстJSON, ПозицияПеревода + 1);
	ТекстJSON = СтрЗаменить(ТекстJSON, "```", "");
	
	Возврат СокрЛП(ТекстJSON);
	
КонецФункции

Функция РазобратьJSONСтроку(СтрокаJSON)
	
	ЧтениеJSON = Новый ЧтениеJSON;
	ЧтениеJSON.УстановитьСтроку(СтрокаJSON);
	Возврат ПрочитатьJSON(ЧтениеJSON, Ложь);
	
КонецФункции

	ЗаписьJSON = Новый ЗаписьJSON;

	ЗаписатьJSON(ЗаписьJSON, Данные);

Функция ОбъединитьРезультатыПроходов(ДанныеШапка, ДанныеТаблица, ИмяТипа)
	
	Результат = Новый Структура;
	Результат.Вставить("version", "2");
	Результат.Вставить("templateType", ИмяТипа);
	Результат.Вставить("header", Новый Структура);
	Результат.Вставить("rows", Новый Массив);
	Результат.Вставить("footer", Новый Структура);
	Результат.Вставить("warnings", Новый Массив);
	Результат.Вставить("errors", Новый Массив);
	
	Если ДанныеШапка <> Неопределено Тогда
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "imageWidth");
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "imageHeight");
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "orientationApplied");
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "qrVisible");
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "referenceDigitsStripVisible");
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "header");
		СкопироватьСвойствоЕслиЕсть(ДанныеШапка, Результат, "footer");
	КонецЕсли;
	
	Если ДанныеТаблица <> Неопределено Тогда
		СкопироватьСвойствоЕслиЕсть(ДанныеТаблица, Результат, "rows");
		Если Не ЗначениеЗаполнено(Результат.orientationApplied) Тогда
			СкопироватьСвойствоЕслиЕсть(ДанныеТаблица, Результат, "orientationApplied");
		КонецЕсли;
	КонецЕсли;
	
	Возврат Результат;
	
КонецФункции

Процедура СкопироватьСвойствоЕслиЕсть(Источник, Приемник, ИмяСвойства)
	
	ЗначениеСвойства = Неопределено;
	Если Источник.Свойство(ИмяСвойства, ЗначениеСвойства) Тогда
		Приемник.Вставить(ИмяСвойства, ЗначениеСвойства);
	КонецЕсли;

	Если Данные = Неопределено Тогда
		Возврат ПустойКаркасОтветаVLM(ИмяТипа);

	ПривестиПлоскуюСхемуККанонической(Данные, ИмяТипа);

	НормализоватьСекциюПолей(Данные, "footer");
	НормализоватьСтрокиТаблицы(Данные);
	ОбеспечитьСлужебныеМассивы(Данные);
	
	Возврат Данные;

Процедура ПривестиTemplateType(Данные, ИмяТипа)
	
	Если Не Данные.Свойство("templateType") Или Не ЗначениеЗаполнено(Данные.templateType) Тогда
		ТипДокумента = Неопределено;
		Если Данные.Свойство("type", ТипДокумента) И ЗначениеЗаполнено(ТипДокумента) Тогда
			Данные.Вставить("templateType", ТипДокумента);
		Иначе
			Данные.Вставить("templateType", ИмяТипа);
		КонецЕсли;
	КонецЕсли;
	
	Если Не Данные.Свойство("version") Тогда
		Данные.Вставить("version", "2");
	КонецЕсли;
	
	Поворот = Неопределено;
	Если Не Данные.Свойство("orientationApplied") И Данные.Свойство("orientation", Поворот) Тогда
		Данные.Вставить("orientationApplied", Поворот);
	КонецЕсли;
	
КонецПроцедуры

Процедура ПривестиПлоскуюСхемуККанонической(Данные, ИмяТипа)
	
	ИменаПолейШапки = ИменаПолейШапкиVLM(ИмяТипа);
	ИменаПолейПодвала = ИменаПолейПодвалаVLM(ИмяТипа);
	
	Если Не Данные.Свойство("header") Тогда
		Данные.Вставить("header", Новый Структура);
	КонецЕсли;
	
	Если Не Данные.Свойство("footer") Тогда
		Данные.Вставить("footer", Новый Структура);
	КонецЕсли;
	
	Если Не Данные.Свойство("rows") Тогда
		Данные.Вставить("rows", Новый Массив);
	КонецЕсли;
	
	Для Каждого ИмяПоля Из ИменаПолейШапки Цикл
		ПеренестиКорневоеПолеВСекцию(Данные, ИмяПоля, "header");
	КонецЦикла;
	
	Для Каждого ИмяПоля Из ИменаПолейПодвала Цикл
		ПеренестиКорневоеПолеВСекцию(Данные, ИмяПоля, "footer");
	КонецЦикла;
	
КонецПроцедуры

Процедура ПеренестиКорневоеПолеВСекцию(Данные, ИмяПоля, ИмяСекции)
	
	ЗначениеПоля = Неопределено;
	Если Не Данные.Свойство(ИмяПоля, ЗначениеПоля) Тогда
		Возврат;
	КонецЕсли;
	
	Секция = Данные[ИмяСекции];
	Если Не Секция.Свойство(ИмяПоля) Тогда
		Секция.Вставить(ИмяПоля, ОбернутьЗначениеВПолеVLM(ИмяПоля, ЗначениеПоля));
	КонецЕсли;
	
	Данные.Удалить(ИмяПоля);
	
КонецПроцедуры

Функция ОбернутьЗначениеВПолеVLM(ИдПоля, Значение)
	
	Поле = Новый Структура;
	Поле.Вставить("id", ИдПоля);
	Поле.Вставить("label", ИдПоля);
	Поле.Вставить("source", "");
	Поле.Вставить("status", ?(ЗначениеЗаполнено(Значение), "ok", "empty"));
	Поле.Вставить("confidence", ?(ЗначениеЗаполнено(Значение), 0.9, 0.1));
	Поле.Вставить("value", Значение);
	
	Возврат Поле;
	
КонецФункции

Процедура НормализоватьСекциюПолей(Данные, ИмяСекции)
	
	Если Не Данные.Свойство(ИмяСекции) Тогда
		Возврат;
	КонецЕсли;
	
	Секция = Данные[ИмяСекции];
	
	Если ТипЗнч(Секция) <> Тип("Структура") Тогда
		Данные.Вставить(ИмяСекции, Новый Структура);
		Возврат;
	КонецЕсли;
	
	Для Каждого КлючЗначение Из Секция Цикл
		Если ТипЗнч(КлючЗначение.Значение) <> Тип("Структура") Тогда
			Секция.Вставить(КлючЗначение.Ключ, ОбернутьЗначениеВПолеVLM(КлючЗначение.Ключ, КлючЗначение.Значение));
		Иначе
			ИсправитьSourceПоля(КлючЗначение.Значение);
		КонецЕсли;
	КонецЦикла;
	
КонецПроцедуры

Процедура ИсправитьSourceПоля(Поле)
	
	ДопустимыеИсточники = Новый Соответствие;
	ДопустимыеИсточники.Вставить("printed", Истина);
	ДопустимыеИсточники.Вставить("handwritten", Истина);
	ДопустимыеИсточники.Вставить("mixed", Истина);
	ДопустимыеИсточники.Вставить("comb", Истина);
	ДопустимыеИсточники.Вставить("signature_present", Истина);
	
	Источник = "";
	Если Поле.Свойство("source", Источник) И ДопустимыеИсточники.Получить(НРег(Источник)) = Неопределено Тогда
		ЗначениеПоля = ЗначениеПоляVLM(Поле);
		Если Не ЗначениеЗаполнено(ЗначениеПоля) И ЗначениеЗаполнено(Источник) Тогда
			Поле.value = Источник;
		КонецЕсли;
		Поле.source = "printed";
	КонецЕсли;

	КонецЕсли;
	
	ВызватьИсключение СтрШаблон(
		НСтр("ru = 'Тип ведомости %1 не поддерживается VLM-промптом.'"),
		ТипВедомости);

	Возврат
	"Ты — OCR/VLM-ассистент для сканов бланков учёта ГСМ (заправочные, приходные, переливные ведомости).
	|
	|Общие правила:
	|1. Сначала мысленно выровняй изображение: заголовок читается горизонтально. Скан может быть повёрнут на 90°/180°/270°; QR — в углу бланка.
	|2. Определи поворот (0/90/180/270). При сомнении — warning ORIENTATION_UNCERTAIN.
	|3. Различай printed / handwritten / empty. Прочерк без рукописного текста = empty.
	|4. Не выдумывай данные: value = null, status = ""unreadable"", confidence < 0.4.
	|5. Полоса «1234567890» — эталон гребёнки; НЕ копируй в таблицу, шапку, итоги. referenceDigitsStripVisible = true если полоса видна.
	|6. Подписи: signature_present | signature_absent; текст подписи не угадывать.
	|7. QR не декодируй — только qrVisible.
	|8. Ответ — ТОЛЬКО валидный JSON UTF-8, без markdown.
	|9. ЗАПРЕЩЕНО копировать одно значение в несколько полей.
	|10. Пустая таблица — rows = []. Не создавай строки с выдуманными данными.
	|11. ФИО не может состоять только из цифр.
	|12. Светло-серые контуры гребёнки без рукописи — empty.
	|13. Номер ведомости — из «Ведомость № …», не из госномера АТЗ.
	|14. СТРОГО соблюдай схему: header — объект полей; rows — [{rowIndex, isEmpty, fields}]; footer — объект полей. НЕ упрощай в плоский JSON.
	|15. Каждое поле — {id, label, source, status, confidence, value, valueString?, cells?}.
	|16. Обойди строки таблицы №1..№16; для каждой строки читай каждую колонку отдельно.
	|17. value — только вписанный/напечатанный ТЕКСТ ЗНАЧЕНИЯ. НЕ возвращай название метки поля («Ф.И.О. водителя», «ПУ:») как value.
	|18. source поля — только ""printed"" | ""handwritten"" | ""mixed"" | ""comb"" | ""signature_present""; не подставляй распознанный текст в source.
	|19. Обязательные корневые ключи: version, templateType, imageWidth, imageHeight, orientationApplied, qrVisible, referenceDigitsStripVisible, header, rows, footer, warnings, errors.
	|20. Печатные реквизиты шапки (organization, statement_number, fuel_truck_plate, fuel_name, date_year) — если текст чётко виден, status = ""ok"", confidence >= 0.9. Не помечай их unreadable без причины.
	|
	|Статусы: ""ok"" | ""partial"" | ""empty"" | ""unreadable"" | ""printed_only""
	|confidence: 0..1
	|
	|Гребёнки: cells[], valueString, value (число или null).
	|Текст: value (строка или null).
	|
	|";
